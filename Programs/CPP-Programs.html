#include <iostream>
#include <stdlib.h>

using namespace std;

// Functions
void inputArray(int Arr[5][5])
{
    for(int i = 0; i < 5; i++)
    {
        for(int j = 0; j < 5; j++)
        {
            Arr[i][j] = rand()%50 + 1; // This Will Generate Random Numbers From 1 to 50.
        }
    }
}

void displayArray(int Arr[5][5])
{
    for(int i = 0; i < 5; i++)
    {
        for(int j = 0; j < 5; j++)
        {
            cout << Arr[i][j] << " ";
        }
        cout << endl;
    }
}

void minMaxArray(int Arr[5][5])
{
    int min, max = Arr[0][0];
    for(int i = 0; i < 5; i++)
    {
        for(int j = 0; j < 5; j++)
        {
            if(Arr[i][j] < min)
            {
                min = Arr[i][j];
            }
            if(Arr[i][j] > max)
            {
                max = Arr[i][j];
            }
        }
    }
    cout << endl;
    cout << "The Minimum Number Is: " << min << endl;
    cout << "The Maximum Number Is: " << max << endl << endl;
}

void evenNumberArray(int Arr[5][5])
{
    cout << endl;
    cout << "Even Numbers In Array Are: " << endl;
    
    for(int i = 0; i < 5; i++)
    {
        for(int j = 0; j < 5; j++)
        {
            if(Arr[i][j]%2 == 0)
            {
                cout << Arr[i][j] << " ";
            }
        }
        cout << endl;
    }
}

void oddNumberArray(int Arr[5][5])
{
    cout << endl;
    cout << "Odd Numbers In Array Are: " << endl;

    {
        for(int j = 0; j < 5; j++)
        {
            if(Arr[i][j]%2 != 0)
            {

}

{

        2D Array Contains Rows and Columns. Syntax: int arrayName[row][columns];

    */

#include <iostream>

using namespace std;

int main ()
{
    
    int num1, num2;
    char operators;
    
    cout << "Enter First Number: ";
    cin >> num1;
    
    cout << "Enter Operator (+ - * / %): ";
    cin >> operators;
    
    cout << "Enter Second Number: ";
    cin >> num2;
    
    switch(operators)
    {
        case '+':
            cout << num1 << " + " << num2 << " = " << (num1 + num2);
            break;
        case '-':
            cout << num1 << " - " << num2 << " = " << (num1 - num2);
            break;
        case '*':
            cout << num1 << " * " << num2 << " = " << (num1 * num2);
            break;
        case '/':
            cout << num1 << " / " << num2 << " = " << (num1 / num2);
            break;
        case '%':
            cout << num1 << " % " << num2 << " = " << (num1 % num2);
            break;	
        default:
            cout << "Invalid Operator. Enter From + - * /";	
            break;
    }
    
    return 0;
}

#include <iostream>

// Varibales
string name;
int rollNo, age, totalMarks;
char grade;
float cgpa;

// Pointers
string *pName = &name;
int *pRollNo = &rollNo, *pAge = &age, *pTotalMarks = &totalMarks;
char *pGrade = &grade;
float *pCgpa = &cgpa;

{
    cout << "Enter Name: ";

    cout << "Enter The Roll No: ";
    cin >> rollNo;
    
    cout << "Enter Age: ";

    cin >> grade;
    
    cout << "Enter CGPA: ";
    cin >> cgpa;
}

{
    cout << "The Name of The Student = " << name << endl;
    cout << "The Address of Name Variable = " << pName << endl << endl;
    
    cout << "The Roll No: of The Student = " << rollNo << endl;
    cout << "The Address of Roll No: Variable = " << pRollNo << endl << endl;
    
    cout << "The Age of The Student = " << age << endl;
    cout << "The Address of Age Variable = " << pAge << endl << endl;
    
    cout << "The Total Marks of The Student = " << totalMarks << endl;
    cout << "The Address of Total Marks Variable = " << pTotalMarks << endl << endl;
    
    cout << "The Grade of The Student = " << grade << endl;
    cout << "The Address of Grade Variable = " << pGrade << endl << endl;
    
    cout << "The CGPA of The Student = " << cgpa << endl;
    cout << "The Address of CGPA Variable = " << pCgpa << endl << endl;
}

int main()
{
    // Pointers are special variables which are used to store address of other variables.
    // Data Type of Pointer Should Be Same as Data Type of Variable.
    
    // Normal Varibale
    int a = 10;
    cout << "Value of Variable A is: " << a << endl;
    cout << "Address of Variable A is: " << &a << endl;
    
    // Pointer
    int *p;
    p = &a;
    
    cout << "Value of Pointer P is: " << p << endl;
    cout << "Value of That Variable Which Address is Stored in Pointer P is: " << *p << endl;
    
    // Assignment
    input();
    display();
    
    return 0;
}

                return true;

    isDraw = true;
    return false;

    for(int i = 1; i <= 10; i++)
    {
        // Spaces Before Stars
        for(int j = 1; j <= 10-i; j++)
        {
            cout << " ";
        }
        // This If Will Print 1 Star and Last Line of Star When I will match condition
        if(i == 1 || i == 10)
        {
            for(int k = 1; k <= i; k++)
            {
                cout << "* ";
            }
        }
        else
        {
            // Left Side Line
            cout << "*";
            
            // Printing Space Inside of Shape
            for(int k = 1; k <= 2*i-3; k++) 
            {
                cout << " ";
            }
