4. Relational & Logical Operators

#include <iostream>
using namespace std;

int main()
{
    int a = 5;
    int b = 2;
    int c = 1;
    int d = 2;
    float result;
    
    // Checking Precedence: (), /, *, -, + 
    result = ( (a + b) - c / d + (c + d) );
    
    cout << "Result: " << result;
    
    // Relational Operators	
    cout << "\n" << a << " == " << b << " = " << (a == b);
    cout << "\n" << a << " > " << b << " = " << (a > b);
    cout << "\n" << a << " < " << b << " = " << (a < b);
    cout << "\n" << a << " >= " << b << " = " << (a >= b);
    cout << "\n" << a << " <= " << b << " = " << (a <= b);
    cout << "\n" << a << " != " << b << " = " << (a != b);
    
    // Logical Opperators
    cout << "\n" <<	(a  < b ) && (a > 1);
    cout << "\n" <<	(a  < b ) || (a > b);
    cout << "\n" <<	!( (a  < b ) || (a > b) );

    return 0;
}

8. Do While and Array

#include <iostream>

using namespace std;

int main()
{
    // Do While Loop
    int a = 1;
    do
    {
        cout << a << endl;
        a++;
        // a = a + 2;
    } while(a <= 10);

    // Custom Series: 1 3 4 6 8 9 10
    for(int i = 1; i <= 10; i++)
    {
        if(i == 2 || i == 5 || i == 7)
        {
            continue;
        }
        else
        {
            cout << i << " ";
        }
    }
    
    // Escape Every 4 Number (Method 01)
    int count = 0;
    for(int j = 1; j <= 50; j++)
    {
        count++;
        
        if(count == 4)
        {
            count = 0;
        }
        else
        {
            cout << j << " ";
        }
    }
    
    // Escape Every 4 Number (Method 02)
    for(int a = 1; a <= 50; a++)
    {
        if(a%4 == 0)
        {
            continue;
        }
        cout << a << " ";
    }
    
    // Array
    int arr[10];
    arr[0] = 11;
    arr[1] = 22;
    arr[2] = 33;
    arr[3] = 44;
    arr[4] = 55;
    arr[5] = 66;
    arr[6] = 77;
    arr[7] = 88;
    arr[8] = 99;
    arr[9] = 100;
    
    cout << arr[2];
    cout << arr[6];
    cout << arr[9];
    
    int index;
    for(index = 0; index < 10; index++)
    {
        cout << arr[index] << endl;
        cout << "arr[" << index << "]" << " = " << arr[index] << endl;
    }
    
    int size;
    cout << "Enter Array Size: ";
    cin >> size;
    
    int arr[size];
    
    int index;
    
    // For Input In Array
    for(index = 0; index < size; index++)
    {
        cout << "Enter Value at Array [" << index << "] = ";
        cin >> arr[index];
        cout << endl;
    }
    
    // For Printing Array
    for(index = 0; index < size; index++)
    {
        cout << arr[index] << endl;
    }
    
    // Size of Array
    int length = sizeof(arr)/sizeof(arr[0]);
    int length = sizeof(arr)/sizeof(arr[0])-1;
    cout << length;
    
    return 0;
}

10. Bitwise Operators

#include <iostream>

using namespace std;

int main()
{
    int a = 20;
    int b = 50;
    int c = 40;
    int res;
    
    /*
    AND &
    
        128		64		32		16		8		4		2		1
    
a =		0		0		0		1		0		1		0		0
b =		0		0		1		1		0		0		1		0
        ---------------------------------------------------------
a&b		0		0		0		1		0		0		0		0
c =		0		0		1		0		1		0		0		0
        ---------------------------------------------------------
(a&b)&c 0		0		0		0		0		0		0		0
    
    Add Those Numbers Where Bit is 1
    */
    
    res = ((a & b) & c);
    cout << res << endl;
    
    /*
    OR |
    
        128		64		32		16		8		4		2		1
    
a =		0		0		0		1		0		1		0		0
b =		0		0		1		1		0		0		1		0
        ---------------------------------------------------------
a|b 	0		0		1		1		0		1		1		0
c =		0		0		1		0		1		0		0		0
        ---------------------------------------------------------
(a|b)&c	0		0		1		0		0		0		0		0
    
    */
    res = ((a | b) & c);
    cout << res << endl;
    
    a = 10;
    /*
    Shift Left	<<
        128		64		32		16		8		4		2		1
a =		0		0		0		0		1		0		1		0
        ---------------------------------------------------------
a<<2	0		0		1		0		1		0		0		0

    */
    res = (a << 2);	
    cout << res << endl;
    
    int x = 10;
    int y = 5;
    cout << (x << y) << endl;
    
    a = 10;
    /*
    Shift Right	>>
        128		64		32		16		8		4		2		1
a =		0		0		0		0		1		0		1		0
        ---------------------------------------------------------
a>>2	0		0		0		0		0		0		1		0		0						

    */
    res = (a >> 2);	
    cout << res << endl;
    
    a = 10;
    b = 40;
    /*	
    XOR ^
    
        128		64		32		16		8		4		2		1
    
a =		0		0		0		0		1		0		1		0
b =		0		0		1		0		1		0		0		0
        ---------------------------------------------------------
a^b 	0		0		1		0		0		0		1		0
    
    */
    res = (a ^ b);
    cout << res << endl;
    
    
    return 0;
}

11. Pointers

#include <iostream>

using namespace std;

int main()
{
    int x = 10;
    
    // Ways of Declaring Pointer
    // int *a;
    int* a;
    
    cout << "The Value of X = " << x << endl;
    cout << "The Address of X = " << &x << endl;
    
    a = &x;
    
    cout << "The Value of A = " << a << endl;
    cout << "The Address of A = " << &a << endl;
    
    // We Are Getting X Value of By Pointer
    cout << "The Value of *A = " << *a << endl;
    
    return 0;
}
